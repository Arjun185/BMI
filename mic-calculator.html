Monthly Interest Calculator

// Monthly Interest Calculator function calculateMonthlyInterest() { const principal = parseFloat(document.getElementById("principal").value); const rate = parseFloat(document.getElementById("rate").value); const fromDate = new Date(document.getElementById("fromDate").value); const toDate = new Date(document.getElementById("toDate").value); if (isNaN(principal) || isNaN(rate) || isNaN(fromDate.getTime()) || isNaN(toDate.getTime())) { alert("Please fill all fields correctly."); return; } const timeDiff = Math.abs(toDate - fromDate); const days = Math.ceil(timeDiff / (1000 * 60 * 60 * 24)); const monthlyInterest = (principal * rate * days) / (30 * 100); document.getElementById("result").innerText = `Monthly Interest: ₹${monthlyInterest.toFixed(2)}`; }
